# 🌏 글로벌 작가 온보딩 페이지 구현안 v2.0

## 📋 프로젝트 개요

### 목적
**아이디어스 글로벌 판매를 원하는 작가님**을 대상으로, 재미있고 직관적인 학습 경험을 제공하는 온보딩 교보재 페이지를 구축합니다.

### 원본 노션 문서
- **메인 가이드**: https://idus.notion.site/d50ff6b383014f1abfed25ccab054007
- **작성일**: 2024-12-22
- **버전**: 2.0 (작가 대상 게임화 버전)

### 핵심 키워드
- 🎮 **게임화 (Gamification)**: 재미있는 학습 경험
- 🚀 **직관적 UX**: 쉽게 따라갈 수 있는 단계별 가이드
- 🏆 **성취감**: 보상과 진행률로 동기 부여
- 🤝 **작가 친화적**: 따뜻하고 응원하는 톤앤매너

## 🎮 게임화(Gamification) 설계

### 1. 레벨 시스템

```
🌱 Lv.1 새싹 작가
    ↓ (STAGE 1 완료)
🌿 Lv.2 성장하는 작가
    ↓ (STAGE 2 완료)
🌳 Lv.3 글로벌 작가
    ↓ (STAGE 3 완료)
🌟 Lv.4 마스터 작가
```

### 2. 배지 시스템

| 배지 | 이름 | 획득 조건 | 설명 |
|------|------|----------|------|
| 🚀 | 첫 발걸음 | 온보딩 시작 | "글로벌 여정을 시작했어요!" |
| 🌐 | 세계 탐험가 | STAGE 1 완료 | "글로벌 서비스를 이해했어요!" |
| 🎨 | 아티스트 | STAGE 2 완료 | "작품 등록 마스터!" |
| 📦 | 배송 프로 | STAGE 3 완료 | "주문부터 배송까지 완벽!" |
| 🏆 | 글로벌 챔피언 | 전체 완료 | "진정한 글로벌 작가!" |
| ⚡ | 스피드러너 | 30분 내 완료 | "빠른 학습의 달인!" |
| 📚 | 학구파 | FAQ 전체 확인 | "궁금증 해결 완료!" |
| 🔥 | 연속 학습 | 3일 연속 접속 | "꾸준함의 힘!" |

// hooks/useProgress.ts
export function useProgress() {
  const [progress, setProgress] = useState<ProgressState>(getInitialProgress);
  
  const markContentComplete = (contentId: string) => {
    setProgress(prev => ({
      ...prev,
      completedContents: [...prev.completedContents, contentId],
      xp: prev.xp + getContentXP(contentId)
    }));
    saveProgress();
    checkBadgeUnlock(contentId);
  };
  
  const getStageProgress = (stageId: string) => {
    const stage = STAGES.find(s => s.id === stageId);
    const completed = stage?.contents.filter(c => 
      progress.completedContents.includes(c)
    ).length || 0;
    return (completed / (stage?.contents.length || 1)) * 100;
  };
  
  const getTotalProgress = () => {
    const total = STAGES.flatMap(s => s.contents).length;
    return (progress.completedContents.length / total) * 100;
  };
  
  return { 
    progress, 
    markContentComplete, 
    getStageProgress, 
    getTotalProgress 
  };
}

// hooks/useGamification.ts
export function useGamification() {
  const [badges, setBadges] = useState<string[]>([]);
  const [level, setLevel] = useState(1);
  const [showCelebration, setShowCelebration] = useState(false);
  
  const unlockBadge = (badgeId: string) => {
    if (!badges.includes(badgeId)) {
      setBadges(prev => [...prev, badgeId]);
      setShowCelebration(true);
      playSound('badge-unlock');
      confetti();
    }
  };
  
  const calculateLevel = (xp: number) => {
    if (xp >= 200) return 4;
    if (xp >= 120) return 3;
    if (xp >= 50) return 2;
    return 1;
  };
  
  const getLevelInfo = (level: number) => {
    const levels = {
      1: { name: '새싹 작가', emoji: '🌱' },
      2: { name: '성장하는 작가', emoji: '🌿' },
      3: { name: '글로벌 작가', emoji: '🌳' },
      4: { name: '마스터 작가', emoji: '🌟' }
    };
    return levels[level] || levels[1];
  };
  
  return { 
    badges, 
    level, 
    unlockBadge, 
    getLevelInfo, 
    showCelebration, 
    setShowCelebration 
  };
}
