# 📚 학습 페이지 중점 리팩토링 제안서 (v4 - 최종)

> **목표**: "번역/소통 어렵지 않아요" 메시지 전달 & 학습 피로도 최소화
> **핵심 타겟**: "글로벌 판매? 번역도 해야 하고 해외 고객 응대도 해야 하는 거 아니야?"
> **분석일**: 2026-01-02

## 🎯 핵심 인사이트

### 작가님들의 실제 우려사항 (우선순위)

| 순위 | 우려사항 | 해소 메시지 |
|------|----------|------------|
| 🥇 **1위** | "번역 어떻게 해요?" | ChatGPT로 1분이면 OK! |
| 🥈 **2위** | "해외 고객이 문의하면?" | 자동 번역 + 한국어 답변 OK! |
| 🥉 **3위** | "해외 배송 복잡하지 않아요?" | 국내 택배 보내듯이 끝! |
| 4위 | "뭘 팔 수 있어요?" | 대부분 OK, 식품/향수만 X |

### 리팩토링 방향 재정립

```
❌ 기존 방향: 모든 콘텐츠에 탭 구조 적용 → 산만함
✅ 수정 방향: 
   - 탭 구조는 최소화 (복잡한 비교 정보에만)
   - 인포그래픽 적극 활용 (고퀄리티, 유치하지 않게)
   - 핵심 우려사항(번역/소통) 해소에 집중
```

'use client';

import { motion } from 'framer-motion';
import { FileText, Bot, CheckCircle, Clock, Lightbulb } from 'lucide-react';

export function TranslationProcessInfographic() {
  const steps = [
    { icon: FileText, label: '한국어', sublabel: '작품 설명', color: 'bg-idus-gray' },
    { icon: Bot, label: 'ChatGPT', sublabel: '붙여넣기', color: 'bg-blue-50 border-blue-200' },
    { icon: CheckCircle, label: '영어', sublabel: '완성!', color: 'bg-green-50 border-green-200' },
  ];

  return (
    <div className="bg-gradient-to-br from-white to-idus-gray/50 rounded-2xl p-6 border border-idus-black-10">
      {/* 헤더 */}
      <div className="text-center mb-6">
        <motion.div 
          className="inline-flex items-center gap-2 bg-idus-orange-light/50 px-4 py-2 rounded-full"
          initial={{ opacity: 0, y: -10 }}
          animate={{ opacity: 1, y: 0 }}
        >
          <Clock className="w-4 h-4 text-idus-orange" />

      </div>

      <div className="flex items-center justify-center gap-2 sm:gap-4">
        {steps.map((step, index) => (
          <motion.div 
            key={index}
            className="flex items-center"
            initial={{ opacity: 0, x: -20 }}
            animate={{ opacity: 1, x: 0 }}
            transition={{ delay: index * 0.15 }}
          >
            {/* 스텝 카드 */}
            <div className={`
              flex flex-col items-center p-4 sm:p-6 rounded-xl border
              ${step.color}
              transition-transform hover:scale-105
            `}>
              <div className="w-12 h-12 sm:w-14 sm:h-14 rounded-full bg-white shadow-sm 
                            flex items-center justify-center mb-3">
                <step.icon className="w-6 h-6 sm:w-7 sm:h-7 text-idus-black-70" />
              </div>
              <div className="text-sm sm:text-base font-bold text-idus-black">{step.label}</div>
              <div className="text-xs text-idus-black-50">{step.sublabel}</div>
            </div>

            {/* 화살표 */}
            {index < steps.length - 1 && (
              <motion.div 
                className="mx-1 sm:mx-3 text-idus-orange text-xl sm:text-2xl"
                animate={{ x: [0, 4, 0] }}
                transition={{ repeat: Infinity, duration: 1.5 }}
              >
                →
              </motion.div>
            )}
          </motion.div>
        ))}
      </div>

      {/* TIP */}
      <motion.div 
        className="mt-6 flex items-start gap-3 bg-amber-50 rounded-xl p-4 border border-amber-200"
        initial={{ opacity: 0 }}
        animate={{ opacity: 1 }}
        transition={{ delay: 0.5 }}
      >
        <Lightbulb className="w-5 h-5 text-amber-500 flex-shrink-0 mt-0.5" />
        <div>
          <span className="font-medium text-amber-700">TIP</span>
          <span className="text-amber-800 ml-2">
            "이 작품 설명을 영어로 번역해줘" 라고만 하면 끝!
          </span>
        </div>
      </motion.div>
    </div>
  );
}

// components/learning/infographics/CustomerSupport.tsx
'use client';

import { motion } from 'framer-motion';
import { MessageCircle, RefreshCw, User, Check } from 'lucide-react';

interface Message {
  flag?: string;
  role: 'customer' | 'translate' | 'artist';
  text: string;
  note?: string;
}

export function CustomerSupportInfographic() {
  const messages: Message[] = [
    { flag: '🇺🇸', role: 'customer', text: '"When will my order arrive?"' },
    { role: 'translate', text: '"주문이 언제 도착하나요?"', note: 'idus가 자동으로!' },
    { flag: '🇰🇷', role: 'artist', text: '"7-14일 정도 소요됩니다"', note: '한국어로 답변!' },
    { role: 'translate', text: '"It takes about 7-14 days"', note: '다시 자동으로!' },
  ];

  const roleStyles = {
    customer: 'bg-blue-50 border-blue-200 ml-0 mr-auto',
    translate: 'bg-idus-orange-light/30 border-idus-orange/30 mx-auto',
    artist: 'bg-green-50 border-green-200 ml-auto mr-0',
  };

  const roleIcons = {
    customer: <MessageCircle className="w-4 h-4" />,
    translate: <RefreshCw className="w-4 h-4" />,
    artist: <User className="w-4 h-4" />,
  };

  return (
    <div className="bg-gradient-to-br from-white to-idus-gray/50 rounded-2xl p-6 border border-idus-black-10">
      {/* 메시지 플로우 */}
      <div className="space-y-3 max-w-sm mx-auto">
        {messages.map((msg, index) => (
          <motion.div
            key={index}
            className="relative"
            initial={{ opacity: 0, y: 20 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ delay: index * 0.2 }}
          >
            <div className={`
              relative p-4 rounded-xl border max-w-[85%]
              ${roleStyles[msg.role]}
            `}>
              {/* 아이콘 + 플래그 */}
              <div className="flex items-center gap-2 mb-1">
                {msg.flag && <span className="text-lg">{msg.flag}</span>}
                <span className="text-idus-black-50">{roleIcons[msg.role]}</span>
              </div>
              
              {/* 텍스트 */}
              <div className="text-sm text-idus-black font-medium">
                {msg.text}
              </div>

              {/* 노트 */}
              {msg.note && (
                <div className="absolute -right-2 top-1/2 -translate-y-1/2 translate-x-full
                              text-xs text-idus-orange font-medium whitespace-nowrap
                              hidden sm:block">
                  ← {msg.note}
                </div>
              )}
            </div>

            {/* 화살표 */}
            {index < messages.length - 1 && (
              <div className="flex justify-center my-1">
                <motion.div 
                  className="text-idus-black-30"
                  animate={{ y: [0, 3, 0] }}
                  transition={{ repeat: Infinity, duration: 1 }}
                >
                  ↓
                </motion.div>
              </div>
            )}
          </motion.div>
        ))}
      </div>

      {/* 하단 배지 */}
      <motion.div 
        className="flex flex-wrap justify-center gap-3 mt-6"
        initial={{ opacity: 0 }}
        animate={{ opacity: 1 }}
        transition={{ delay: 1 }}
      >
        {['영어', '일본어', '중국어'].map((lang) => (
          <div key={lang} className="flex items-center gap-1 bg-white px-3 py-1.5 rounded-full
                                    border border-idus-black-10 text-sm">
            <Check className="w-4 h-4 text-green-500" />
            <span className="text-idus-black">{lang} 몰라도 OK!</span>
          </div>
        ))}
      </motion.div>
    </div>
  );
}
